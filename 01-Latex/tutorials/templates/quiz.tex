\begin{problem}{10}
Prove that there exist no positive integers $x$, $y$, and $z$ such that
\[
x^3 + y^3 = z^3.
\]
\end{problem}

\begin{problem}{15}

\begin{problem}{20}
Let $f \colon \mathbf{N} \to \mathbf{N}$ be given by
\[
f(n) :=
\begin{cases}
3n+1 & \text{if $n$ is odd}, \\
n/2 & \text{if $n$ is even}.
\end{cases}
\]
For any $n \in \mathbf{N}$, do we have some $k \in \mathbf{N}$ such that
$f^{(k)}(n) = 1$? Justify your answer.
\end{problem}
